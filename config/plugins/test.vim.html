 4   autocmd BufEnter term://* startinsert

 7 fu! ExistsTerminal(tnmb, tname)
 8     redir => buffers
 9     silent ls
10     redir END
11     for buf in split(buffers, '\n')
12         if match(buf, '\v^\s*'.a:tnmb) > -1 && match(buf, a:tname) > -1
13             return 1
14         endif
15     endfor
16     return 0
17 endf

19 " 终端是否显示
20 fu! ActiveTerminal(tnmb, tname)
21     redir => buffers
22     silent ls
23     redir END
24     for buf in split(buffers, '\n')
25         if match(buf, '\v^\s*'.a:tnmb) > -1 && match(buf, a:tname) > -1
26             " 有a即活动
27             if match(split(buf)[1], "a-") > -1 || match(split(buf)[1], "a") > -1
28                 return 1
29             else
30                 return 0
31             endif
32         endif
33     endfor
34     return 0
35 endf

38 let s:bnr = -1
39 let s:bna = ""
40 let s:winnr = -1
41 " 底部打开终端
42 fu! OpenTerminalBottom()
43     botright split
44     resize 15
45     if ExistsTerminal(s:bnr, s:bna)
46         exe "b".s:bnr
47     else
48         if has('nvim')
49             exec "terminal"
50         else
51             :call term_start('bash', {'curwin' : 1, 'term_finish' : 'close'})
52         endif
53         exe "normal! a"
54         let s:bnr = bufnr("%")
55         let s:bna = bufname("%")
56         let s:winnr = winnr()
57         echom s:winnr
58     endif
59 endf
60 
61 fu! OpenTerminalBottomToggle()
62     " 如果活动
63     if ActiveTerminal(s:bnr, s:bna)
64         " 跳转到那个窗口
65         " exec s:winnr . 'wincmd w'
66         " 关闭这个窗口
67         exec "close ".s:winnr
68     else
69         :call OpenTerminalBottom()
70     endif
71 endf
